package Algorytmy_cwiczenia;

public class PD_Fibo {
    public static void main(String[] args) {
        int n = 6;
        System.out.println("Fibonachi n = " + n + " jest r√≥wny " + fibonachi(n));
    }
    public static int fibonachi(int n) {
        int[] fib = new int[n + 1];
        fib[0] = 0;
        fib[1] = 1;
        for (int i = 2 ; i <= n; i++) {
            fib[i] = fib[i - 1] + fib[i - 2];
        }
        return fib[n];
    }
}
